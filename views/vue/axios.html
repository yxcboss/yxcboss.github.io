        # axios概述

        在vue项目中，和后台交互数据这块，我们通常使用的是axios库。

        # axios功能特点

        • 在浏览器中发送XMLHttpRequests请求
        • 在node.js中发送http请求
        • 支持Prosime API
        • 拦截请求和响应
        • 。。。

        # axios的使用与封装

        # 安装

        npm istall axios
        1

        # 引入

        在项目中一般会在src目录中，新建一个request文件夹，在里面新建一个http.js和一个api.js文件。http.js用来封装我们的axios，api.js用来统一管理我们的接口。

        # 环境的切换 ，设置axios的默认请求地址

        我们项目中会有开发环境，测试环境和生产环境。我们通过node的环境变量来匹配默认接口的url前缀，axios。default.baseURL可以设置axios的默认请求地址。

        /* 
 * 根据环境变量进行接口区分
 */
switch (process.env.NODE_ENV) {
    case "development":
        axios.defaults.baseURL = "";
        break;
    case "test":
        axios.defaults.baseURL = "";
        break;
    case "production":
        axios.defaults.baseURL = "";
        break;
}

        # 设置请求超时

        通过axios.default.timeout设置默认的请求超时时间。例如超过了10s，就会告知用户当前请求超时，请刷新等。

        axios.default.timeout = 10000
        1

        # 设置CORS跨域允许携带资源凭证

        axios.defaults.withCredentials = true;
        1

        # 设置post请求头，告知服务器请求主题的数据格式

        axios.defaults.headers.post['Content-Type'] = 'application/x-www-form-urlencoded;charset=UTF-8';

        # 请求拦截

        axios.interceptors.request.use(
    config=>{
        // 添加TOKEN验证：可以从本地存储中取值，也可以从VUEX中取值，主要看怎么存储的
        //如果存在，则统一在http请求的header都加上token
        const token = localStorage.getItem('token');
        token && (config.headers.Authorization = token);
        return config; 
    },
    error=>{
        return Promise.error(error)
    }
)

        # 响应拦截

        axios.interceptors.response.use(
    response=>{
        //如果返回的状态码是200，说明请求成功，否则抛出错误
        if(response.status === 200){
            return Promise.resolve(response);
        }else{
            return Promise.reject(response);
        }
    },
    error=>{
        if(error.response.status){
            switch (error.response.status) {
                case 401: // 当前请求需要用户验证（一般是未登录）
                    break;
                case 403: // 服务器已经理解请求，但是拒绝执行它（一般是TOKEN过期）
                    localStorage.removeItem('token');
                    // 跳转到登录页
                    break;
                case 404: // 请求失败，请求所希望得到的资源未被在服务器上发现
                    break;
                default: //否则抛出错误
                    alert(error.response.data.msg);
            }
            return Promise.reject(error.response);
        }else{
            // 未发送请求，断网处理
            return Promise.reject(error.response);
        }
    }
)

        # 封装get和post方法

        /**
 * get方法，对应get请求
 * @param {String} url [请求的url地址]
 * @param {Object} params [请求时携带的参数]
 */ 
export get(url,params)=>{
    return new Promise((reject,reject)=>{
        axios.get(url,{
            params:params
        }).then(res=>{
            resolve(res.data);
        }).catch(err=>{
            reject(err.data);
        })
    })
}

// post方法需要对提交的数据进行序列化的操作，所有引用qs模块，来序列化post类型的数据
/** 
 * post方法，对应post请求 
 * @param {String} url [请求的url地址] 
 * @param {Object} params [请求时携带的参数] 
 */
export post(url,params)=>{
    return new Promise((resolve,reject)=>{
        axios.post(url,QS.stringify(params)).then(res=>{
            resolve(res.data);
        }).catch(err=>{
            reject(err.data);
        })
    })
}

        # api的统一管理

        • 在api.js中引入我们封装的get和post方法
        • 定义一个接口方法，接收一个参数params，然后调用我们引入的get或者post方法，最后通过export导出。
        import {get,post} from './http'
const _api = {
    apiIndex:'course/apiIndex'
    // ... 
}
export apiIndex(params)=>{
    get(_api.apiIndex,params)
}
