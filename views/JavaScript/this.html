              # 总结：this的绑定规则分为5种。

              1. 默认绑定（严格模式与非严格模式）
              2. 隐式绑定
              3. 显式绑定
              4. new绑定
              5. 箭头函数绑定

              # 默认绑定

              • 独立函数调用，this指向全局对象，即Window
                  var name = 'yxc'
    function fun(){
        console.log(this,this.name)
        // Window，yxc
    }
    fun();

              this被谁调用就指向谁

    let obj = {

        }

              # 显示绑定

              通过call()、apply()或者bind()方法，在函数调用时，绑定到指定对象上

                  var name = 'wll'
    function say(){
        console.log(this.name);
    }
    var obj = {
        name:'yxc',
        say:say
    }
    obj.say();
    obj.say.call(window);
    obj.say.apply(window);
    obj.say.bind(window)();

              # new绑定

              如果构造函数没有返回函数或者对象，那么this指向这个构造函数生成的对象

                  function Person(name,age){
        this.name=name;
        this.age=age;
        this.sayHi=function(){
            console.log(this)
        }
    }
    let person=new Person("俞玄赐",18);
    person.sayHi();
    // Person {name: "俞玄赐", age: 6, sayHi: ƒ}

              # 箭头函数绑定

              箭头函数没有自己的this，继承外层上下文绑定的this

                  let name = 'wll'
    let obj = {
        name:'yxc',
        say:()=>{
            console.log(this,this.name);
            //由于箭头函数没有自己的this，所以绑定到上层的window，this.name为wll
        }
    }
    obj.say();

              # 题外话

              # call，apply，bind三者的区别

              都是用于改变this的指向，但用法略有不同。call和apply的第一个参数为this的只指向，而后面所传参数形式不同。call第二个参数及以上都是当前函数的参数，而apply为所有参数组成的数组。而bind传参方式与call相同，不过改变指向时，需要在后面加个()调用,类似于函数。示例代码于显示绑定
